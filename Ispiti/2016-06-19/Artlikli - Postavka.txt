struct Datum 
{
    int * _dan, *_mjesec, *_godina;
    void Unos(int d, int m, int g) 
    {
        _dan = new int(d);
        _mjesec = new int(m);
        _godina = new int(g);
    }
    void Ispis() { cout << *_dan << "/" << *_mjesec << "/" << *_godina << endl; }
    void Dealociraj() 
    {
        delete _dan; delete _mjesec; delete _godina;
        _dan = _mjesec = _godina = nullptr;
    }
};
